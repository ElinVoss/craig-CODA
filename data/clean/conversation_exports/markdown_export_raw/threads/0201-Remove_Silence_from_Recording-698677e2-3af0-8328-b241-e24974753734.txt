def merge_ranges(ranges, gap_ms=200):
    if not ranges: return []
    merged=[ranges[0][:]]
    for start,end in ranges[1:]:
        prev=merged[-1]
        if start - prev[1] <= gap_ms:
            prev[1]=max(prev[1], end)
        else:
            merged.append([start,end])
    return merged

import numpy as np, math

def segment_features(sig, sr):
    # sig: float -1..1
    # RMS dB
    rms=np.sqrt(np.mean(sig**2)+1e-12)
    rms_db=20*np.log10(rms+1e-12)
    peak=np.max(np.abs(sig))+1e-12
    peak_db=20*np.log10(peak)
    clip_ratio=float(np.mean(np.abs(sig)>0.98))
    # DC offset
    dc=float(np.mean(sig))
    # spectral flatness & band ratios
    # downsample to 16000 for analysis
    target_sr=16000
    if sr!=target_sr:
        # simple decimate via slicing using rational factor if divisible
        factor=sr//target_sr
        if sr%target_sr==0:
            sig_ds=sig[::factor]
            sr_ds=target_sr
        else:
            # fallback: linear interpolation
            idx=np.arange(0,len(sig))*target_sr/sr
            idx2=np.arange(int(len(sig)*target_sr/sr))
            sig_ds=np.interp(idx2, idx, sig).astype(np.float32)
            sr_ds=target_sr
    else:
        sig_ds=sig
        sr_ds=sr
    # take up to first 2 seconds for spectral stats; or whole if shorter
    max_n=int(sr_ds*2.0)
    y=sig_ds[:max_n] if len(sig_ds)>max_n else sig_ds
    if len(y)<256:
        flat=1.0
        band_ratio=0.0
        hiss_ratio=0.0
    else:
        # apply hann window
        win=np.hanning(len(y))
        Y=np.fft.rfft(y*win)
        mag=np.abs(Y)+1e-12
        # spectral flatness = geometric_mean/arithmetic_mean
        flat=float(np.exp(np.mean(np.log(mag))) / (np.mean(mag)+1e-12))
        freqs=np.fft.rfftfreq(len(y), 1/sr_ds)
        # speech band 300-3400
        speech_band=(freqs>=300)&(freqs<=3400)
        total=np.sum(mag)
        band=np.sum(mag[speech_band])
        band_ratio=float(band/(total+1e-12))
        # hiss band >6000
        hiss=(freqs>=6000)&(freqs<=sr_ds/2)
        hiss_ratio=float(np.sum(mag[hiss])/(total+1e-12))
    return {
        "rms_db": float(rms_db),
        "peak_db": float(peak_db),
        "clip_ratio": clip_ratio,
        "dc": dc,
        "flatness": flat,
        "band_ratio": band_ratio,
        "hiss_ratio": hiss_ratio,
    }

def segment_features(sig, sr):
    rms=np.sqrt(np.mean(sig**2)+1e-12)
    rms_db=20*np.log10(rms+1e-12)
    peak=np.max(np.abs(sig))+1e-12
    peak_db=20*np.log10(peak)
    clip_ratio=float(np.mean(np.abs(sig)>0.98))
    dc=float(np.mean(sig))
    # downsample to 16000 for spectral stats
    target_sr=16000
    if sr==target_sr:
        sig_ds=sig
        sr_ds=sr
    else:
        if sr%target_sr==0:
            factor=sr//target_sr
            sig_ds=sig[::factor]
            sr_ds=target_sr
        else:
            # linear resample
            n_out=int(len(sig)*target_sr/sr)
            t_in=np.linspace(0,1,len(sig),endpoint=False)
            t_out=np.linspace(0,1,n_out,endpoint=False)
            sig_ds=np.interp(t_out, t_in, sig).astype(np.float32)
            sr_ds=target_sr
    max_n=int(sr_ds*2.0)
    y=sig_ds[:max_n] if len(sig_ds)>max_n else sig_ds
    if len(y)<256:
        flat=1.0
        band_ratio=0.0
        hiss_ratio=0.0
    else:
        win=np.hanning(len(y))
        Y=np.fft.rfft(y*win)
        mag=np.abs(Y)+1e-12
        flat=float(np.exp(np.mean(np.log(mag))) / (np.mean(mag)+1e-12))
        freqs=np.fft.rfftfreq(len(y), 1/sr_ds)
        total=np.sum(mag)
        band=np.sum(mag[(freqs>=300)&(freqs<=3400)])

        hiss=np.sum(mag[(freqs>=6000)&(freqs<=sr_ds/2)])
        hiss_ratio=float(hiss/(total+1e-12))
    return dict(rms_db=float(rms_db), peak_db=float(peak_db), clip_ratio=clip_ratio, dc=dc,
