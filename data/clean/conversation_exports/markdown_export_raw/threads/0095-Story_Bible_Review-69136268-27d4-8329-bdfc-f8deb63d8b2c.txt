# Inspect and extract the uploaded zip, then build a file tree and previews for text files.

import os, zipfile, textwrap, json, pandas as pd
from pathlib import Path
from caas_jupyter_tools import display_dataframe_to_user

zip_path = "/mnt/data/ConfluenceChronicles-StoryBible-main.zip"
extract_dir = "/mnt/data/ConfluenceChronicles"

# Ensure clean extract dir
if os.path.exists(extract_dir):
    # Don't delete; just re-use
    pass
else:
    os.makedirs(extract_dir, exist_ok=True)

# Extract
with zipfile.ZipFile(zip_path, 'r') as zf:
    zf.extractall(extract_dir)

# Build file tree metadata
rows = []
text_previews = []
text_exts = {".md", ".txt", ".json", ".yaml", ".yml", ".ini", ".config", ".cfg", ".csv", ".tsv", ".toml"}

for root, dirs, files in os.walk(extract_dir):
    for f in files:
        p = Path(root) / f
        rel = str(p.relative_to(extract_dir))
        size = p.stat().st_size
        ext = p.suffix.lower()
        rows.append({
            "relative_path": rel,
            "ext": ext,
            "size_bytes": size
        })
        if ext in text_exts and size <= 2_000_000:  # limit preview to 2MB files
            try:
                content = p.read_text(encoding="utf-8", errors="ignore")
                preview = content[:1500]
                text_previews.append({
                    "relative_path": rel,
                    "preview": preview
                })
            except Exception as e:
                text_previews.append({
                    "relative_path": rel,
                    "preview": f"[Error reading file: {e}]"
                })

# Display file inventory
df_files = pd.DataFrame(rows).sort_values("relative_path").reset_index(drop=True)
display_dataframe_to_user("Archive Contents: ConfluenceChronicles", df_files)

# Display previews
df_previews = pd.DataFrame(text_previews).sort_values("relative_path").reset_index(drop=True)
display_dataframe_to_user("Text Previews (first ~1500 chars)", df_previews)

def find_bible_for_n(novella_dir: Path):
    # Priority order: files with "bible" in name, else README.md, else any .md
    md_files = list(novella_dir.rglob("*.md"))
    if not md_files:
        return None
    # rank by filename patterns
    def rank(p: Path):
        name = p.name.lower()
        score = 0
        if "bible" in name: score += 100
        if name in ("readme.md", "index.md"): score += 50
        # prefer shorter paths (closer to dir root)
        score -= len(p.parts)
        return score
    md_files.sort(key=rank, reverse=True)
    return md_files[0]

def extract_sections(text: str):

        sections[title] = body
    # Known keys to look for

# Build concise summaries for each novella by extracting synopsis-like text and trimming to ~120 words.

def find_bible_for_n(novella_dir: Path):
    md_files = list(novella_dir.rglob("*.md"))
    if not md_files:
        return None
    def rank(p: Path):
        name = p.name.lower()
        score = 0
        if "bible" in name: score += 100
        if name in ("readme.md", "index.md"): score += 50
        score -= len(p.parts)
        return score
    md_files.sort(key=rank, reverse=True)
    return md_files[0]

def extract_sections(text: str):
    t = text.replace("\r\n","\n").replace("\r","\n")
    headings = list(re.finditer(r"(?m)^(#{1,6})\s*(.+?)\s*$", t))
    sections = {}
    for i, h in enumerate(headings):
        start = h.end()
        end = headings[i+1].start() if i+1 < len(headings) else len(t)
        sections[h.group(2).strip().lower()] = t[start:end].strip()
    keys = ["synopsis","summary","overview","premise","hook","logline","story","outline"]
    for k in keys:
        for title, body in sections.items():
            if k in title:
                return body
    # fallback: first paragraph
    paras = [p.strip() for p in t.split("\n\n") if p.strip()]
    return paras[0] if paras else ""

def strip_md(s: str):
    s = re.sub(r"`{1,3}[^`]+`{1,3}", "", s)  # inline code
    s = re.sub(r"!\[[^\]]*\]\([^)]+\)", "", s)  # images
    s = re.sub(r"\[[^\]]+\]\([^)]+\)", r"", s)  # links
    s = re.sub(r"[*_#>`~-]{1,}", " ", s)  # markup chars
    s = re.sub(r"\s+", " ", s).strip()
    return s

def truncate_words(s: str, max_words=120):
    words = s.split()
    return " ".join(words[:max_words]) + ("…" if len(words) > max_words else "")

# Extract concrete bullet points like "Major Events (IMMUTABLE)" and metadata (Turning, Position) to build factual summaries.
import re, json
from pathlib import Path
import pandas as pd
from caas_jupyter_tools import display_dataframe_to_user

root = Path("/mnt/data/NovellaBibles")
files = sorted([p for p in root.glob("N??_Bible_*.md")])

def parse_file(p: Path):
    text = p.read_text(encoding="utf-8", errors="ignore")
    # Metadata lines
    turning = re.search(r"\*\*Turning\*\*:\s*([^\n]+)", text)
    position = re.search(r"\*\*Position\*\*:\s*([^\n]+)", text)
    innocence = re.search(r"\*\*Innocence Protocol\*\*:\s*([^\n]+)", text)
    meta = {
        "turning": turning.group(1).strip() if turning else "",
        "position": position.group(1).strip() if position else "",
        "innocence_protocol": innocence.group(1).strip() if innocence else ""
    }
    # Find sections with "Major Events" (case-insensitive)
    events = []
    for m in re.finditer(r"(?mi)^##\s*.*major events.*?\n((?:.|\n)*?)(?=^##\s|\Z)", text):
        block = m.group(1)
        for b in re.finditer(r"(?m)^\s*[-*]\s+(.*)", block):
            item = b.group(1).strip()
            # stop at overly long items
            events.append(item)
    # Also catch bullet lines tagged (IMMUTABLE) elsewhere
    if not events:
        for m in re.finditer(r"(?mi)^\s*[-*]\s+(.*\(IMMUTABLE\).*)$", text):
            events.append(m.group(1).strip())
    # Fallback: first 5 bullets anywhere under "NOVELLA SCOPE" or "SCOPE"
    if not events:
        for m in re.finditer(r"(?mi)^##\s*.*scope.*?\n((?:.|\n)*?)(?=^##\s|\Z)", text):
            block = m.group(1)
            for b in re.finditer(r"(?m)^\s*[-*]\s+(.*)", block):
                events.append(b.group(1).strip())
            if events:
                break
    # Trim to first 8 items
    events = events[:8]
    return meta, events

records = []
for p in files:
    m = re.match(r"(N\d{2})_Bible_(.+)\.md", p.name)
    nid = m.group(1)
    title = m.group(2).replace("_"," ")
    meta, events = parse_file(p)
    records.append({
        "novella": nid,
        "title": title,
        "turning": meta["turning"],
        "position": meta["position"],
        "innocence_protocol": meta["innocence_protocol"],
        "events": events
    })

# Compose concise paragraph summaries from the parsed events for each novella and save as markdown plus emit in output.
import json, re
from pathlib import Path

parsed_path = Path("/mnt/data/Parsed_Bible_Events.json")
records = json.loads(parsed_path.read_text(encoding="utf-8"))

def clean_event(e):
    # Remove markdown emphasis and brackets
    e = re.sub(r"\*{1,2}","", e)
    e = re.sub(r"\[ ?\] ?", "", e)  # checkbox
    e = e.replace("—", "-")
    return e.strip()

lines = ["# Confluence Chronicles — Brief Summaries by Novella", ""]
for r in sorted(records, key=lambda x: x["novella"]):
    nid = r["novella"]
    title = r["title"].title()
    turning = r.get("turning","").strip()
    position = r.get("position","").strip()
    ev = [clean_event(e) for e in r.get("events", [])]
    # Build paragraph: include up to 5 events
    if ev:
        core = "; ".join(ev[:5])
        para = f"{nid} — {title}: {core}."
    else:
        meta = []
        if turning: meta.append(f"Turning: {turning}")
        if position: meta.append(position)
        meta_text = " | ".join(meta) if meta else "No concrete events present."
        para = f"{nid} — {title}: {meta_text}"
    lines.append(para)
md = "\n\n".join(lines)
out_path = "/mnt/data/Novella_Brief_Summaries_FOR_CHAT.md"
Path(out_path).write_text(md, encoding="utf-8")

# Extract and print "Major Events" lines for a few specific bibles to reduce ambiguity.
import re
from pathlib import Path

root = Path("/mnt/data/NovellaBibles")

def print_major_events(fname):
    p = root / fname
    print(f"== {fname} ==")
    text = p.read_text(encoding="utf-8", errors="ignore")
    # Find blocks with "Major Events"
    for m in re.finditer(r"(?mi)^###?\s*.*major events.*?\n((?:.|\n)*?)(?=^##\s|\Z)", text):
        block = m.group(1)
        for b in re.finditer(r"(?m)^\s*[-*]\s+(.*)", block):
            print("- " + b.group(1).strip())
    # Also scan for "Chapter" bullets in general
    for b in re.finditer(r"(?m)^\s*[-*]\s+(Chapter [^:\n]+:[^\n]+)", text):
        print("- " + b.group(1).strip())

    if (root/fname).exists():
        print_major_events(fname)
    else:
        print(f"{fname} missing")

# Extract chapter-event bullets for all bibles and print per-novella concise summaries from those bullets.
import re
from pathlib import Path

root = Path("/mnt/data/NovellaBibles")
files = sorted([p for p in root.glob("N??_Bible_*.md")])

def chapter_events(text):
    ev = []
    # find sections labeled major events
    for m in re.finditer(r"(?mi)^###?\s*.*major events.*?\n((?:.|\n)*?)(?=^##\s|\Z)", text):
        block = m.group(1)
        for b in re.finditer(r"(?m)^\s*[-*]\s+(Chapter [^:\n]+:[^\n]+)", block):
            ev.append(b.group(1).strip())
    # if none, generic scan for bullets starting with Chapter
    if not ev:
        for b in re.finditer(r"(?m)^\s*[-*]\s+(Chapter [^:\n]+:[^\n]+)", text):
            ev.append(b.group(1).strip())
    # dedupe and cap
    seen = set()
    out = []
    for e in ev:
        if e not in seen:
            seen.add(e)
            out.append(e)
    return out[:7]

for p in files:
    text = p.read_text(encoding="utf-8", errors="ignore")
    nid = p.name.split("_",1)[0]
    title = p.name.split("_Bible_",1)[1].rsplit(".",1)[0].replace("_"," ").title()
    ev = chapter_events(text)
    print(f"==== {nid} — {title}")
    if not ev:
        print("(no explicit Chapter bullets found)")
    else:
        for i, e in enumerate(ev, 1):
            print(f"{i}. {e}")

### N01 — Grip That Holds  *(GLASS 1/5)*
A planet-piercing impact blows a 30km hole through Te’Oga. Jhace accidentally stabilizes a collapsing scaffold using dual-principle magic, drawing Tiffani (a form-lock master) to the crisis. They partner up, begin training, suffer learning-deaths, weather a brutal followshock, and finally commit—to each other and to a nascent crew. The heat-grid is largely restored, vein-shatter recedes, and distant reports whisper of “god-vessels” awakening elsewhere.

### N02 — Light Cracks Through  *(GLASS 2/5)*
Vein-shatter stress ticks up; followshocks become routine. Independent practitioners emerge beyond Jhace/Tiffani’s circle as Marisse steps forward as an organizing voice. As the pair coordinates a growing group, ideology and ambition start to rub: Tiffani pushes for scale, Jhace fears overreach. A Soulpulse-overuse death hammers home the doctrine of cost and cracks their relationship.

### N03 — Alloy Folded Wrong  *(GLASS 3/5)*
Practitioners turn up poisoned with vein-burns that don’t match known accidents—malice is in play. Smuggling routes and corrupted practice are exposed as Marisse’s influence (and friction with Tiffani) grows. The conflict escalates to fatal consequences and a direct confrontation with the bad actors. It ends in grief, a sober recommitment to defense and teaching.

### N04 — Shatter From Heat  *(GLASS 4/5)*
Hollowing cases appear and accelerate; the cost of power becomes visible in bodies. Coordinated attacks strike civic and energy systems; the city staggers. Jhace and Tiffani hold the line, but the pressure exposes secrets and fractures. He confronts her about the mystery she carries; the partnership strains under the heat.

### N05 — Steal Breaks Open  *(GLASS 5/5 → CRACK begins)*
Reality destabilizes—“Steal Breaks Open” replaces vein-shatter and cascades out of control. Jhace manifests impossible **Confluence** capacity (all four principles at once) as a desperate response, not destiny. **Tiffani dies** during the event (reads as tragic/heroic, not orchestrated). The four gods fully awaken, the world’s order revolts, and Jhace is left alone, carrying the Confluence.

### N06 — Called From Refuge  *(CRACK 1/5)*
In the wreckage, Jhace turns to teaching to metabolize grief; first students gather. Marisse arrives and pushes the work forward in Tiffani’s memory. A grassroots resistance network takes shape as distant signs of god-vessel faction formation appear. Tiffani’s absence and legacy define the emotional runway.

### N07 — Resonance Awakens  *(CRACK 2/5)*
The four god-vessel factions formalize and start competing for followers. Civilians are forced to pick sides or get crushed between them. Infrastructure becomes a weapon; sabotage starts. Jhace refuses all gods, and resistance hardens into a fourth force. Marisse’s body shows the bill for decades of use.

### N08 — Adaptations Cost  *(CRACK 3/5)*
Marisse keeps teaching even as Hollowing worsens; the cost doctrine is now a face and a tremor. God-vessels escalate and shift tactics. Marisse hits irreversible cascade and **dies**; Rennik steps up as chief medic. Jhace accepts that there’s no clean path—only debt management.

### N09 — Creations Test  *(CRACK 4/5)*
Despite losses, the school keeps running; new practitioners need structure more than ever. First real attempts to **create** durable solutions under fire begin—and fail, then half-work. The team absorbs more grief; Jessa moves into greater responsibility. Teaching persists because it must.

### N10 — Knowledge Demanded  *(CRACK 5/5 → FORGE opens)*
Organized sabotage ramps—knowledge becomes the high ground. The gods demand doctrinal fealty; defectors start to appear seeking protection. The resistance’s logistics and intel networks stretch to breaking. The line between “teacher” and “commander” is erased.

### N11 — Forging the Sacred  *(FORGE 1/5)*
All four god-vessels elevate their leaders to near-sacred status and claim moral monopoly. First god-on-god clashes erupt. A civic inspector, Harven, dies as collateral—an indictment of everyone’s methods. The ideological war is no longer hypothetical.

### N12 — Order Sought  *(FORGE 2/5)*
Purge operations and release-doctrine displacements create a refugee crisis. Witnesses Halrick (Mobel methods) and Mourne (Eries doctrine) testify; the scale is undeniable. A building collapse kills civilians; Mrs. Kettner’s death forces Jhace to accept limits: he cannot save everyone.

### N13 — Resistance Grows  *(FORGE 3/5)*
The god-vessel factions escalate strategically; the resistance matures into a full military structure. Fights with lieutenants on all sides test resolve and doctrine. The resistance is now the fourth faction in fact, not name.

### N14 — Gods Entrench  *(FORGE 4/5)*
The gods stop poaching from each other and fortify—territories calcify. Defectors trickle out with actionable intel; Brighlee organizes a civic backbone to keep normal people alive. The board looks set for a siege.

### N15 — Escalation Burns  *(FORGE 5/5 → SMELT opens)*
The siege hits. Jhace coordinates a multi-front defense; **Kylar** makes a final stand, buys evacuation with his life. **Daric** wages psy-ops with release doctrine. The line holds; victory is pyrrhic; FORGE closes and SMELT—grinding war—begins.
