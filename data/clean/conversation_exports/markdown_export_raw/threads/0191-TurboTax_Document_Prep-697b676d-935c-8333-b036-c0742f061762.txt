import numpy as np

def deskew_angle(im, angle_range=(-3,3), step=0.1):
    gray = im.convert('L')
    # downsample for speed
    small = gray.resize((600, int(600*gray.size[1]/gray.size[0])), Image.Resampling.BICUBIC)
    best_angle=0
    best_score=-1
    for a in np.arange(angle_range[0], angle_range[1]+1e-9, step):
        rot = small.rotate(a, resample=Image.Resampling.BICUBIC, expand=True, fillcolor=255)
        arr=np.array(rot)
        # binarize
        bw = arr < 200
        # compute horizontal projection (sum black pixels per row)
        proj = bw.sum(axis=1)
        # score as variance of differences (peaky lines)
        score = np.var(np.diff(proj))
        if score>best_score:
            best_score=score
            best_angle=a
    return best_angle, best_score

from PIL import ImageEnhance, ImageOps, ImageFilter

def enhance_scan(im):
    g = im.convert('L')
    # increase contrast
    g = ImageOps.autocontrast(g, cutoff=1)
    g = ImageEnhance.Contrast(g).enhance(1.6)
    g = ImageEnhance.Brightness(g).enhance(1.05)
    # sharpen
    g = g.filter(ImageFilter.UnsharpMask(radius=2, percent=180, threshold=3))
    return g

import numpy as np

def binarize(im):
    arr=np.array(im.convert('L'))
    # Otsu threshold
    hist, bins = np.histogram(arr.flatten(), 256, [0,256])
    total=arr.size
    sum_total=np.dot(np.arange(256), hist)
    sumB=0
    wB=0
    varMax=0
    threshold=128
    for t in range(256):
        wB += hist[t]
        if wB==0: 
            continue
        wF = total - wB
        if wF==0:
            break
        sumB += t*hist[t]
        mB = sumB / wB
        mF = (sum_total - sumB) / wF
        varBetween = wB*wF*(mB-mF)**2
        if varBetween>varMax:
            varMax=varBetween
            threshold=t
    bw = (arr > threshold).astype(np.uint8)*255
    return Image.fromarray(bw, mode='L')
