        root = tree.getroot()

        data_rows = []
        for row in root.findall('.//ss:Row', ns):
            for cell in row.findall('ss:Cell', ns):
                data_node = cell.find('ss:Data', ns)
                if data_node is not None and data_node.text:
                    lines = data_node.text.split('\n')
                    for line in lines:
                        if line.startswith('|') and 'StorageBin' not in line and '---' not in line:
                            parts = [p.strip() for p in line.split('|')]
                            # The header suggests:

                            # 1: Typ
                            # 2: Sec
                            # 3: P
                            # 4: R
                            # 5: E
                            # 6: StorageBin
                            # 7: Materia
                            # 8: Material Description
                            # 9: Avail.stock
                            # 10: BUn
                            # 11: Total Stock
                            # ...
                            # 17: Total capacity
                            # 18: Remain.capacity

                                        'Description': description,
                                        'TotalStock': total_stock,
                                        'Capacity': capacity
                                    })
                                except (ValueError, IndexError):

        return pd.DataFrame(data_rows)
    except Exception as e:

import sys
import os
import re
import pandas as pd
import xml.etree.ElementTree as ET
import tkinter as tk
from tkinter import filedialog, messagebox, ttk

class BinConsolidatorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Plasti-Pak Line Clearance Optimizer")
        self.root.geometry("800x600")
        
        self.file_path = tk.StringVar()
        self.results_text = None
        
        self.setup_ui()
        
    def setup_ui(self):
        main_frame = ttk.Frame(self.root, padding="10")
        main_frame.pack(fill=tk.BOTH, expand=True)
        
        file_frame = ttk.LabelFrame(main_frame, text="Input SAP Snapshot", padding="10")
        file_frame.pack(fill=tk.X, pady=5)
        
        ttk.Entry(file_frame, textvariable=self.file_path, width=70).pack(side=tk.LEFT, padx=5)
        ttk.Button(file_frame, text="Browse", command=self.browse_file).pack(side=tk.LEFT)
        
        btn_frame = ttk.Frame(main_frame, padding="10")
        btn_frame.pack(fill=tk.X)
        
        ttk.Button(btn_frame, text="Generate Extraction List", command=self.process_data).pack(side=tk.LEFT, padx=5)
        ttk.Button(btn_frame, text="Save Move List", command=self.save_results).pack(side=tk.LEFT, padx=5)
        
        results_frame = ttk.LabelFrame(main_frame, text="Extraction List (Goal: Clear Bins 47 -> 02)", padding="10")
        results_frame.pack(fill=tk.BOTH, expand=True, pady=5)
        
        self.results_text = tk.Text(results_frame, wrap=tk.WORD)
        self.results_text.pack(fill=tk.BOTH, expand=True)
        
        scrollbar = ttk.Scrollbar(self.results_text, orient=tk.VERTICAL, command=self.results_text.yview)
        scrollbar.pack(side=tk.RIGHT, fill=tk.Y)
        self.results_text.configure(yscrollcommand=scrollbar.set)

    def browse_file(self):
        filename = filedialog.askopenfilename(filetypes=[("XML files", "*.xml"), ("All files", "*.*")])
        if filename:
            self.file_path.set(filename)

    def parse_sap_xml(self, file_path):
        try:
            tree = ET.parse(file_path)
            root = tree.getroot()
            ns = {'ss': 'urn:schemas-microsoft-com:office:spreadsheet'}
            data_rows = []
            for row in root.findall('.//ss:Row', ns):
                for cell in row.findall('ss:Cell', ns):
                    data_node = cell.find('ss:Data', ns)
                    if data_node is not None and data_node.text:
                        lines = data_node.text.split('\n')
                        for line in lines:
                            if line.startswith('|') and 'StorageBin' not in line and '---' not in line:
                                parts = [p.strip() for p in line.split('|')]
                                if len(parts) > 11:
                                    try:
                                        bin_name = parts[6]
                                        material = parts[7]
                                        description = parts[8]
                                        total_stock = float(parts[11].replace(',', '')) if parts[11] else 0.0
                                        capacity = float(parts[17].replace(',', '')) if len(parts) > 17 and parts[17] else 0.0
                                        data_rows.append({
                                            'Bin': bin_name,
                                            'Material': material,
                                            'Description': description,
                                            'TotalStock': total_stock,
                                            'Capacity': capacity
                                        })
                                    except (ValueError, IndexError):
                                        continue
            return pd.DataFrame(data_rows)
        except Exception as e:
            messagebox.showerror("Error", f"Failed to parse XML: {e}")
            return None

    def process_data(self):
        path = self.file_path.get()
        if not path:
            messagebox.showwarning("Warning", "Please select a file first.")
            return
        df = self.parse_sap_xml(path)
        if df is None or df.empty: return
        
        pattern = re.compile(r'^([A-G])(0[2-9]|[1-3][0-9]|4[0-7])$')
        df['Match'] = df['Bin'].apply(lambda x: pattern.match(x))
        df_filtered = df[df['Match'].notnull()].copy()
        if df_filtered.empty:
            messagebox.showinfo("Info", "No bins found in the A02-G47 range.")
            return

        df_filtered['Row'] = df_filtered['Bin'].str[0]
        df_filtered['Num'] = df_filtered['Bin'].str[1:].astype(int)
        
        moves = []
        for material, group in df_filtered.groupby('Material'):
            if len(group) < 2: continue
            
            # EXTRACTION LOGIC:
            # Source: Bins with HIGHEST Num (closest to line)
            # Target: Bins with LOWEST Num (further from line)
            
            # Sort group to find sources (highest Num first)
            sources = group[group['TotalStock'] > 0].sort_values(by='Num', ascending=False)
            # Sort group to find targets (lowest Num first)
            targets = group.sort_values(by='Num', ascending=True)
            
            for s_idx, s_row in sources.iterrows():
                source_bin = s_row['Bin']
                source_num = s_row['Num']
                source_stock = s_row['TotalStock']
                
                for t_idx, t_row in targets.iterrows():
                    target_bin = t_row['Bin']
                    target_num = t_row['Num']
                    
                    # Only move from high number to lower number
                    if target_num >= source_num: continue
                    
                    target_cap = t_row['Capacity']
                    target_stock = t_row['TotalStock']
                    
                    if target_stock >= target_cap: continue
                    
                    space = target_cap - target_stock
                    move_qty = min(source_stock, space)
                    
                    if move_qty > 0:
                        moves.append({
                            'Material': material,
                            'Desc': s_row['Description'],
                            'From': source_bin,
                            'To': target_bin,
                            'Qty': move_qty
                        })
                        source_stock -= move_qty
                        targets.at[t_idx, 'TotalStock'] += move_qty
                        
                    if source_stock <= 0: break
        
        self.display_results(moves)

    def display_results(self, moves):
        self.results_text.delete(1.0, tk.END)
        if not moves:
            self.results_text.insert(tk.END, "No extraction moves found. Bins near the line are already clear or no space further back.")
            return
        self.results_text.insert(tk.END, f"Found {len(moves)} extraction moves to clear the line:\n\n")
        self.results_text.insert(tk.END, f"{'MATERIAL':<12} | {'FROM (LINE)':<11} -> {'TO (BACK)':<10} | {'QTY':<6} | {'DESCRIPTION'}\n")
        self.results_text.insert(tk.END, "-" * 90 + "\n")
        for m in moves:
            line = f"{m['Material']:<12} | {m['From']:<11} -> {m['To']:<10} | {m['Qty']:<6.2f} | {m['Desc']}\n"
            self.results_text.insert(tk.END, line)

    def save_results(self):
        content = self.results_text.get(1.0, tk.END)
        if not content.strip(): return
        filename = filedialog.asksaveasfilename(defaultextension=".txt", filetypes=[("Text files", "*.txt")])
        if filename:
            with open(filename, 'w') as f: f.write(content)
            messagebox.showinfo("Success", "Extraction list saved.")

if __name__ == "__main__":
    root = tk.Tk()
    app = BinConsolidatorApp(root)
    root.mainloop()

from tkinter import filedialog, messagebox, ttk

class BinConsolidatorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Plasti-Pak Line Clearance Optimizer")
        self.root.geometry("800x600")
        
        self.file_path = tk.StringVar()
        self.results_text = None
        
        self.setup_ui()
        
    def setup_ui(self):
        main_frame = ttk.Frame(self.root, padding="10")
        main_frame.pack(fill=tk.BOTH, expand=True)
        
        file_frame = ttk.LabelFrame(main_frame, text="Input SAP Snapshot", padding="10")
        file_frame.pack(fill=tk.X, pady=5)
        
        ttk.Entry(file_frame, textvariable=self.file_path, width=70).pack(side=tk.LEFT, padx=5)
        ttk.Button(file_frame, text="Browse", command=self.browse_file).pack(side=tk.LEFT)
        
        btn_frame = ttk.Frame(main_frame, padding="10")
        btn_frame.pack(fill=tk.X)
        
        ttk.Button(btn_frame, text="Generate Extraction List", command=self.process_data).pack(side=tk.LEFT, padx=5)
        ttk.Button(btn_frame, text="Save Move List", command=self.save_results).pack(side=tk.LEFT, padx=5)
        
        results_frame = ttk.LabelFrame(main_frame, text="Extraction List (Goal: Clear Bins 47 -> 02)", padding="10")
        results_frame.pack(fill=tk.BOTH, expand=True, pady=5)
        
        self.results_text = tk.Text(results_frame, wrap=tk.WORD)
        self.results_text.pack(fill=tk.BOTH, expand=True)
        
        scrollbar = ttk.Scrollbar(self.results_text, orient=tk.VERTICAL, command=self.results_text.yview)
        scrollbar.pack(side=tk.RIGHT, fill=tk.Y)
        self.results_text.configure(yscrollcommand=scrollbar.set)

    def browse_file(self):
        filename = filedialog.askopenfilename(filetypes=[("XML files", "*.xml"), ("All files", "*.*")])
        if filename:
            self.file_path.set(filename)

    def parse_sap_xml(self, file_path):
        try:
            tree = ET.parse(file_path)
            root = tree.getroot()
            ns = {'ss': 'urn:schemas-microsoft-com:office:spreadsheet'}
            data_rows = []
            for row in root.findall('.//ss:Row', ns):
                for cell in row.findall('ss:Cell', ns):
                    data_node = cell.find('ss:Data', ns)
                    if data_node is not None and data_node.text:
                        lines = data_node.text.split('\n')
                        for line in lines:
                            if line.startswith('|') and 'StorageBin' not in line and '---' not in line:
                                parts = [p.strip() for p in line.split('|')]
                                if len(parts) > 11:
                                    try:
                                        bin_name = parts[6]
                                        material = parts[7]
                                        description = parts[8]
                                        total_stock = float(parts[11].replace(',', '')) if parts[11] else 0.0
                                        capacity = float(parts[17].replace(',', '')) if len(parts) > 17 and parts[17] else 0.0
                                        data_rows.append({
                                            'Bin': bin_name,
                                            'Material': material,
                                            'Description': description,
                                            'TotalStock': total_stock,
                                            'Capacity': capacity
                                        })
                                    except (ValueError, IndexError):
                                        continue
            return pd.DataFrame(data_rows)
        except Exception as e:
            messagebox.showerror("Error", f"Failed to parse XML: {e}")
            return None

    def process_data(self):
        path = self.file_path.get()
        if not path:
            messagebox.showwarning("Warning", "Please select a file first.")
            return
        df = self.parse_sap_xml(path)
        if df is None or df.empty: return
        
        pattern = re.compile(r'^([A-G])(0[2-9]|[1-3][0-9]|4[0-7])$')
        df['Match'] = df['Bin'].apply(lambda x: pattern.match(x))
        df_filtered = df[df['Match'].notnull()].copy()
        if df_filtered.empty:
            messagebox.showinfo("Info", "No bins found in the A02-G47 range.")
            return

        df_filtered['Row'] = df_filtered['Bin'].str[0]
        df_filtered['Num'] = df_filtered['Bin'].str[1:].astype(int)
        
        moves = []
        for material, group in df_filtered.groupby('Material'):
            if len(group) < 2: continue
            
            # EXTRACTION LOGIC:
            # Source: Bins with HIGHEST Num (closest to line)
            # Target: Bins with LOWEST Num (further from line)
            
            # Sort group to find sources (highest Num first)
            sources = group[group['TotalStock'] > 0].sort_values(by='Num', ascending=False)
            # Sort group to find targets (lowest Num first)
            targets = group.sort_values(by='Num', ascending=True)
            
            for s_idx, s_row in sources.iterrows():
                source_bin = s_row['Bin']
                source_num = s_row['Num']
                source_stock = s_row['TotalStock']
                
                for t_idx, t_row in targets.iterrows():
                    target_bin = t_row['Bin']
                    target_num = t_row['Num']
                    
                    # Only move from high number to lower number
                    if target_num >= source_num: continue
                    
                    target_cap = t_row['Capacity']
                    target_stock = t_row['TotalStock']
                    
                    if target_stock >= target_cap: continue
                    
                    space = target_cap - target_stock
                    move_qty = min(source_stock, space)
                    
                    if move_qty > 0:
                        moves.append({
                            'Material': material,
                            'Desc': s_row['Description'],
                            'From': source_bin,
                            'To': target_bin,
                            'Qty': move_qty
                        })
                        source_stock -= move_qty
                        targets.at[t_idx, 'TotalStock'] += move_qty
                        
                    if source_stock <= 0: break
        
        self.display_results(moves)

    def display_results(self, moves):
        self.results_text.delete(1.0, tk.END)
        if not moves:
            self.results_text.insert(tk.END, "No extraction moves found. Bins near the line are already clear or no space further back.")
            return
        self.results_text.insert(tk.END, f"Found {len(moves)} extraction moves to clear the line:\n\n")
        self.results_text.insert(tk.END, f"{'MATERIAL':<12} | {'FROM (LINE)':<11} -> {'TO (BACK)':<10} | {'QTY':<6} | {'DESCRIPTION'}\n")
        self.results_text.insert(tk.END, "-" * 90 + "\n")
        for m in moves:
            line = f"{m['Material']:<12} | {m['From']:<11} -> {m['To']:<10} | {m['Qty']:<6.2f} | {m['Desc']}\n"
            self.results_text.insert(tk.END, line)

    def save_results(self):
        content = self.results_text.get(1.0, tk.END)
        if not content.strip(): return
        filename = filedialog.asksaveasfilename(defaultextension=".txt", filetypes=[("Text files", "*.txt")])
        if filename:
            with open(filename, 'w') as f: f.write(content)
            messagebox.showinfo("Success", "Extraction list saved.")

if __name__ == "__main__":
    root = tk.Tk()
    app = BinConsolidatorApp(root)
    root.mainloop()

SAP’s “Consolidate Stock → Maximize No. of Free Bins” is not something that appears magically for every user in every SAP system; it’s a specific EWM transaction (`/SCWM/STOCK_CONS`) that must be installed, configured and authorised.  A community guide makes it clear that the function is started via **Extended Warehouse Management → Work Scheduling → Consolidate Stock**, or by calling transaction **/SCWM/STOCK_CONS**, and that in a Fiori launchpad it appears under the “EWM – Work Scheduling – Consolidate Stock” app【929282840597509†L410-L414】.  The same guide explains that no special customizing is required, but that proper EWM configuration must already be in place and that it’s normally a supervisor who runs the transaction and creates warehouse tasks【929282840597509†L418-L427】.

If no one in your warehouse has ever mentioned this process, there are a few possibilities:

- **Your system might not include EWM.**  The consolidate‑stock function is part of Extended Warehouse Management; it isn’t available in classic LE‑WM or the new Stock Room Management, so if your company hasn’t licensed or implemented EWM, the transaction won’t exist.

- **It could be configured but restricted to certain roles.**  Running `/SCWM/STOCK_CONS` and creating warehouse tasks is typically a supervisor-level activity【929282840597509†L418-L427】.  If your user ID lacks the relevant EWM authorization objects, you won’t see the option.

- **The feature may be unused or unknown.**  Even with EWM installed, organizations sometimes continue to consolidate stock manually because they’re unaware of the standard process or have not trained staff on it.

Your best next step is to talk with your SAP support team or warehouse management about whether your system includes EWM and whether the **Consolidate Stock** transaction is available.  They can check if transaction `/SCWM/STOCK_CONS` exists in your environment, add it to your role if appropriate, and provide guidance on how to use it.  If your company is still on classic LE‑WM or Stock Room Management, then implementing the automatic consolidation you described would require either upgrading to EWM or building a custom solution similar to the one we discussed earlier.

Transaction **LS24** is part of the classic LE‑WM module.  As the SAP help for “Displaying WM Stock Overview” explains, LS24 simply shows which material (article) is located in which storage location and bin【598530442250948†screenshot】; it doesn’t provide any automation for moving stock or consolidating bins.  The automated consolidation process we’ve been discussing (transaction **/SCWM/STOCK_CONS**, launched via *Extended Warehouse Management → Work Scheduling → Consolidate Stock → Maximize No. of Free Bins*) is available only in **SAP Extended Warehouse Management (EWM)**【929282840597509†L410-L414】.  If your warehouse primarily uses LS24, it likely means you’re running the older LE‑WM or Stock Room Management, where that EWM function simply doesn’t exist.

In that case you have two options:

- **Enable EWM.**  Your organisation would need to implement EWM (either embedded in S/4HANA or decentralized) and assign you a role with access to `/SCWM/STOCK_CONS`【929282840597509†L418-L427】.  Once EWM is in place, the system can propose consolidation moves and free up bins automatically.

- **Custom solution.**  If moving to EWM isn’t feasible, you would need to continue with manual bin consolidation or develop an in‑house tool (like the Python/GUI solution we discussed) to analyse LS24 data and generate transfer orders.  LE‑WM itself doesn’t offer a stock‑consolidation transaction.

So using LS24 alone won’t give you automated consolidation; it’s just a reporting transaction【598530442250948†screenshot】.  You’d need EWM or a custom development to achieve the bin‑freeing workflow we’ve been talking about.
