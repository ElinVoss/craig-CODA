    <script>
        // --- Main Application Class ---
        class SpeechToTextApp {
            
            // Constants
            APP_NAME = "AdvancedTranscriber";
            STORAGE_KEY_SETTINGS = this.APP_NAME + "_settings";
            STORAGE_KEY_HISTORY = this.APP_NAME + "_history";
            
            // State
            transcriptionHistory = [];
            recognition = null;
            isListening = false;
            currentTranscription = '';
            
            // DOM Elements (Initialized later)
            elements = {};

            constructor() {
                this.init();
            }

            async init() {
                this.initializeElements();
                this.loadSettings();
                this.loadHistory();
                this.setupEventListeners();
                this.checkBrowserSupport();
            }

            // --- Initialization and Setup ---
            
            initializeElements() {
                // Settings
                this.elements.speechLanguageSelect = document.getElementById('speechLanguage');
                this.elements.fontSizeSelect = document.getElementById('fontSize');
                this.elements.themeSelect = document.getElementById('theme');
                
                // Controls
                this.elements.startBtn = document.getElementById('startBtn');
                this.elements.stopBtn = document.getElementById('stopBtn');
                this.elements.clearCurrentBtn = document.getElementById('clearCurrentBtn');
                this.elements.statusDot = document.getElementById('statusDot');
                this.elements.transcriptionOutput = document.getElementById('transcriptionOutput');
                this.elements.loadingOverlay = document.getElementById('loadingOverlay');
                this.elements.errorContainer = document.getElementById('errorContainer');
                this.elements.successContainer = document.getElementById('successContainer');
                
                // History
                this.elements.historyPanel = document.getElementById('historyPanel');
                this.elements.historyList = document.getElementById('historyList');
                this.elements.toggleHistory = document.getElementById('toggleHistory');
                this.elements.historyToggleText = document.getElementById('historyToggleText');
                this.elements.exportAllBtn = document.getElementById('exportAllBtn');
            }
            
            setupEventListeners() {
                this.elements.startBtn.addEventListener('click', () => this.startListening());
                this.elements.stopBtn.addEventListener('click', () => this.stopListening());
                this.elements.clearCurrentBtn.addEventListener('click', () => this.clearCurrentTranscript());
                this.elements.themeSelect.addEventListener('change', () => this.updateTheme());
                this.elements.fontSizeSelect.addEventListener('change', () => this.updateFontSize());
                this.elements.speechLanguageSelect.addEventListener('change', () => this.updateRecognitionSettings());
                this.elements.toggleHistory.addEventListener('click', () => this.toggleHistoryPanel());
                this.elements.exportAllBtn.addEventListener('click', () => this.exportAllTranscripts());
            }

            checkBrowserSupport() {
                if (!('webkitSpeechRecognition' in window || 'SpeechRecognition' in window)) {
                    this.showError("Your browser does not support the Web Speech API. Transcription will not work.");
                    this.elements.startBtn.disabled = true;
                } else {
                    this.setupRecognition();
                }
            }

            // --- Core Transcription Logic ---

            setupRecognition() {
                // Use webkitSpeechRecognition for broader compatibility
                this.recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
                this.recognition.continuous = true;
                this.recognition.interimResults = true;
                
                this.updateRecognitionSettings();

                this.recognition.onstart = () => {
                    this.isListening = true;
                    this.updateUIState();
                };

                this.recognition.onerror = (event) => {
                    console.error('Speech Recognition Error:', event);
                    this.showError(`Microphone Error: ${event.error}. Please check your microphone and permissions.`);
                    this.stopListening(false); // Stop, but don't try to save
                };

                this.recognition.onend = () => {
                    if (this.isListening) {
                        // If it ends unexpectedly, restart to maintain continuous session
                        console.log("Recognition ended unexpectedly. Restarting...");
                        this.recognition.start();
                    }
                };

                this.recognition.onresult = (event) => {
                    let interimTranscript = '';
                    let finalTranscriptChunk = '';
                    
                    for (let i = event.resultIndex; i < event.results.length; ++i) {
                        const transcript = event.results[i][0].transcript;
                        if (event.results[i].isFinal) {
                            // Add a period and space to final chunks for readability
                            finalTranscriptChunk += transcript + '. ';
                        } else {
                            interimTranscript += transcript;
                        }
                    }
                    
                    // 1. Update Current Transcription (internal state)
                    this.currentTranscription += finalTranscriptChunk;
                    
                    // 2. Update UI
                    let displayText = this.currentTranscription.trim();
                    if (interimTranscript) {
                        // Show the interim text in brackets at the end
                        displayText += ' [Interim: ' + interimTranscript + ']';
                    }
                    
                    this.elements.transcriptionOutput.value = displayText;
                    this.elements.transcriptionOutput.scrollTop = this.elements.transcriptionOutput.scrollHeight;
                };
            }

            startListening() {
                if (!this.recognition) {
                    this.showError("Cannot start: Speech Recognition not supported or initialized.");
                    return;
                }
                
                this.updateRecognitionSettings();
                this.recognition.start();
                this.isListening = true;
                this.updateUIState();
                this.showSuccess("Listening started. Speak now.");
            }

            stopListening(saveOnStop = true) {
                if (!this.isListening) return;
                
                // Stop the recognition service immediately
                this.recognition.stop(); 
                this.isListening = false;
                this.updateUIState();
                
                if (saveOnStop && this.currentTranscription.trim().length > 0) {
                    this.saveTranscription(this.currentTranscription.trim());
                    this.currentTranscription = ''; // Clear current after saving
                    this.elements.transcriptionOutput.value = '';
                } else if (!saveOnStop) {
                    this.showSuccess("Listening stopped. Current transcript was NOT saved.");
                } else {
                    this.showError("Listening stopped. No text was recorded to save.");
                }
            }

            clearCurrentTranscript() {
                if (this.isListening) {
                    this.showError("Cannot clear while actively listening. Please stop listening first.");
                    return;
                }
                if (this.currentTranscription.length > 0) {
                     // Modal replacement for confirmation
                     const confirmClear = document.createElement('div');
                     confirmClear.className = "fixed inset-0 bg-black/75 flex justify-center items-center z-50";
                     confirmClear.innerHTML = `
                        <div class="bg-gray-800 p-6 rounded-lg text-white max-w-sm">
                            <h3 class="text-xl font-bold mb-4">Confirm Clear</h3>
                            <p>Are you sure you want to clear the current transcript? It will be lost if not saved.</p>
                            <div class="flex justify-end gap-3 mt-4">
                                <button id="cancelClear" class="bg-gray-500 hover:bg-gray-600 text-white font-bold py-2 px-4 rounded">Cancel</button>
                                <button id="confirmClear" class="bg-red-600 hover:bg-red-700 text-white font-bold py-2 px-4 rounded">Clear Anyway</button>
                            </div>
                        </div>
                     `;
                     document.body.appendChild(confirmClear);
                     
                     document.getElementById('cancelClear').onclick = () => confirmClear.remove();
                     document.getElementById('confirmClear').onclick = () => {
                        this.currentTranscription = '';
                        this.elements.transcriptionOutput.value = '';
                        this.showSuccess("Current transcript cleared.");
                        confirmClear.remove();
                     };
                }
            }
            
            // --- Local Storage History Functions ---

            loadHistory() {
                try {
                    const historyJson = localStorage.getItem(this.STORAGE_KEY_HISTORY);
                    this.transcriptionHistory = historyJson ? JSON.parse(historyJson) : [];
                    this.renderHistory();
                } catch (e) {
                    console.error("Error loading history:", e);
                    this.showError("Failed to load local history.");
                }
            }
            
            saveHistory() {
                try {
                    localStorage.setItem(this.STORAGE_KEY_HISTORY, JSON.stringify(this.transcriptionHistory));
                } catch (e) {
                    console.error("Error saving history:", e);
                    this.showError("Failed to save history to local storage. Storage limit may be reached.");
                }
            }

            saveTranscription(text) {
                const newId = Date.now().toString(36) + Math.random().toString(36).substr(2); // Simple unique ID
                const newItem = {
                    id: newId,
                    transcript: text,
                    language: this.elements.speechLanguageSelect.value,
                    timestamp: new Date().toISOString(),
                };
                
                this.transcriptionHistory.unshift(newItem); // Add to the start
                this.saveHistory();
                this.renderHistory();
                this.showSuccess(`Session saved successfully to local history.`);
            }
            
            renderHistory() {
                this.elements.historyList.innerHTML = ''; // Clear previous list
                
                if (this.transcriptionHistory.length === 0) {
                    this.elements.historyList.innerHTML = `
                        <div class="empty-history text-center p-4">
                            <p class="opacity-70">No transcriptions saved yet. Start speaking, then click "Stop & Save Session" to save your first transcript.</p>
                        </div>
                    `;
                    return;
                }
                
                this.transcriptionHistory.forEach((item) => {
                    const date = new Date(item.timestamp).toLocaleString();
                    const snippet = item.transcript.substring(0, 150) + (item.transcript.length > 150 ? '...' : '');
                    
                    const itemDiv = document.createElement('div');
                    itemDiv.className = 'history-item flex flex-col md:flex-row justify-between items-start md:items-center space-y-2 md:space-y-0';
                    itemDiv.innerHTML = `
                        <div class="flex-grow min-w-0">
                            <p class="text-sm opacity-60">${date} (${item.language})</p>
                            <p class="text-lg font-medium whitespace-pre-wrap">${snippet}</p>
                        </div>
                        <div class="history-actions flex gap-2">
                            <button class="btn bg-blue-500 border-blue-500 hover:bg-blue-600 text-white text-xs py-1 px-3" onclick="window.app.viewTranscript('${item.id}')">View Full</button>
                            <button class="btn bg-red-500 border-red-500 hover:bg-red-600 text-white text-xs py-1 px-3" onclick="window.app.deleteTranscript('${item.id}')">Delete</button>
                        </div>
                    `;
                    this.elements.historyList.appendChild(itemDiv);
                });
            }
            
            viewTranscript(id) {
                const item = this.transcriptionHistory.find(i => i.id === id);
                if (!item) {
                    this.showError("Transcript not found.");
                    return;
                }
                
                const modal = document.createElement('div');
                modal.className = "fixed inset-0 bg-black/75 flex justify-center items-center z-50";
                modal.innerHTML = `
                   <div class="bg-gray-800 p-6 rounded-lg text-white max-w-4xl w-11/12 max-h-[90vh] overflow-auto">
                       <h3 class="text-2xl font-bold mb-4">Full Transcript Details</h3>
                       <p class="text-sm opacity-60 mb-4">Saved on: ${new Date(item.timestamp).toLocaleString()} (Language: ${item.language})</p>
                       <textarea class="w-full text-lg p-3 rounded bg-gray-700 text-white resize-none" rows="15" readonly>${item.transcript}</textarea>
                       <div class="flex justify-end gap-3 mt-4">
                           <button onclick="document.execCommand('copy')" class="bg-blue-500 hover:bg-blue-600 text-white font-bold py-2 px-4 rounded">Copy Text</button>
                           <button id="closeModal" class="bg-gray-500 hover:bg-gray-600 text-white font-bold py-2 px-4 rounded">Close</button>
                       </div>
                   </div>
                `;
                document.body.appendChild(modal);
                document.getElementById('closeModal').onclick = () => modal.remove();
                // Select the text for easy copying, although the button does the actual copy operation
                document.querySelector('textarea').select(); 
            }

            deleteTranscript(id) {
                // Modal replacement for confirmation
                const confirmDelete = document.createElement('div');
                confirmDelete.className = "fixed inset-0 bg-black/75 flex justify-center items-center z-50";
                confirmDelete.innerHTML = `
                   <div class="bg-gray-800 p-6 rounded-lg text-white max-w-sm">
                       <h3 class="text-xl font-bold mb-4">Confirm Deletion</h3>
                       <p>Are you sure you want to permanently delete this transcript from local storage?</p>
                       <div class="flex justify-end gap-3 mt-4">
                           <button id="cancelDelete" class="bg-gray-500 hover:bg-gray-600 text-white font-bold py-2 px-4 rounded">Cancel</button>
                           <button id="confirmDelete" class="bg-red-600 hover:bg-red-700 text-white font-bold py-2 px-4 rounded">Delete</button>
                       </div>
                   </div>
                `;
                document.body.appendChild(confirmDelete);

                document.getElementById('cancelDelete').onclick = () => confirmDelete.remove();
                document.getElementById('confirmDelete').onclick = () => {
                    this.transcriptionHistory = this.transcriptionHistory.filter(item => item.id !== id);
                    this.saveHistory();
                    this.renderHistory();
                    this.showSuccess("Transcript deleted successfully.");
                    confirmDelete.remove();
                };
            }
            
            // --- Utility Functions ---
            
            toggleHistoryPanel() {
                const isHidden = this.elements.historyPanel.classList.toggle('hidden');
                this.elements.historyToggleText.textContent = isHidden ? 'Show History' : 'Hide History';
            }
            
            exportAllTranscripts() {
                if (this.transcriptionHistory.length === 0) {
                    this.showError("No history to export.");
                    return;
                }
                
                const exportContent = this.transcriptionHistory.map(item => 
                    `--- Session: ${new Date(item.timestamp).toLocaleString()} (${item.language}) ---\n${item.transcript}\n`
                ).join('\n\n');
                
                const filename = `All-Transcripts-${new Date().toISOString().substring(0, 10)}.txt`;
                this.downloadFile(exportContent, filename, 'text/plain');
                this.showSuccess(`All ${this.transcriptionHistory.length} sessions exported as ${filename}.`);
            }

            downloadFile(data, filename, type) {
                const blob = new Blob([data], { type: type });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = filename;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
            }
            
            // Settings and Theme
            loadSettings() {
                try {
                    const settingsJson = localStorage.getItem(this.STORAGE_KEY_SETTINGS);
                    const settings = settingsJson ? JSON.parse(settingsJson) : {};
                    
                    if (settings.theme) {
                        this.elements.themeSelect.value = settings.theme;
                    }
                    if (settings.fontSize) {
                        this.elements.fontSizeSelect.value = settings.fontSize;
                    }
                    if (settings.language) {
                        this.elements.speechLanguageSelect.value = settings.language;
                    }
                    
                    // Apply loaded settings or defaults
                    this.updateTheme(); 
                    this.updateFontSize();
                } catch (e) {
                    console.error("Error loading settings:", e);
                }
            }

            saveSettings() {
                const settings = {
                    theme: this.elements.themeSelect.value,
                    fontSize: this.elements.fontSizeSelect.value,
                    language: this.elements.speechLanguageSelect.value,
                };
                localStorage.setItem(this.STORAGE_KEY_SETTINGS, JSON.stringify(settings));
            }

            updateTheme() {
                const selectedTheme = this.elements.themeSelect.value;
                document.body.className = `theme-${selectedTheme}`;
                this.saveSettings();
            }

            updateFontSize() {
                const selectedSize = this.elements.fontSizeSelect.value;
                this.elements.transcriptionOutput.style.fontSize = selectedSize;
                this.saveSettings();
            }
            
            updateRecognitionSettings() {
                if (this.recognition) {

    exportAllAsSrt(){
        if (!this.transcriptionHistory || this.transcriptionHistory.length===0){ this.showError("No history to export."); return; }
        const allSrtParts = [];
        this.transcriptionHistory.forEach((item, idx)=>{
            const srt = this.convertItemToSrt(item);
            allSrtParts.push(`--- Session ${idx+1}: ${new Date(item.timestamp).toLocaleString()} (${item.language}) ---\n${srt}`);
        });
        const filename = `All-Transcripts-${new Date().toISOString().slice(0,10)}.srt`;
        this.downloadFile(allSrtParts.join('\n\n'), filename, 'text/plain');
        this.showSuccess(`Exported SRT as ${filename}.`);

    convertItemToSrt(item){
        // If segments provided with timestamps: use them to estimate relative times
        if (item.segments && item.segments.length>0){
            // Normalize to first segment time as t0
            const t0 = item.segments[0].ts;
            // produce entries with 4s minimum per segment but allow longer segments if text longer
            const entries = [];
            for (let i=0;i<item.segments.length;i++){
                const seg = item.segments[i];
                const start = Math.max(0, Math.floor((seg.ts - t0)/1000));
                // approximate duration: if there is a next segment, use its start - this start; else base 3s per 20 chars
                let end;
                if (i+1<item.segments.length) end = Math.max(start+1, Math.floor((item.segments[i+1].ts - t0)/1000));
                else {
                    // estimate based on characters
                    const approxSec = Math.max(2, Math.ceil(seg.text.length / 20));
                    end = start + approxSec;
                }
                entries.push({start,end,text:seg.text});
            }
            // format to srt
            return entries.map((e, i)=>`${i+1}\n${this.formatTimeSrt(e.start)} --> ${this.formatTimeSrt(e.end)}\n${e.text}`).join('\n\n');
        } else {
            // No segments: fall back to slicing transcript into 5-second chunks (approx)
            const words = item.transcript.split(/\s+/);
            const wordsPerChunk = 10; // heuristic
            const chunks = [];
            for (let i=0;i<words.length;i+=wordsPerChunk){
                chunks.push(words.slice(i,i+wordsPerChunk).join(' '));
            }
            let sIdx = 0;
            return chunks.map((c, i)=>{
                const start = sIdx;
                const end = sIdx + 4;
                sIdx += 5;
                return `${i+1}\n${this.formatTimeSrt(start)} --> ${this.formatTimeSrt(end)}\n${c}`;
            }).join('\n\n');
        }
    }

    formatTimeSrt(sec){
        // sec: integer seconds
        const h = Math.floor(sec/3600); const m = Math.floor((sec%3600)/60); const s = sec%60;
        const ms = 0;
        const pad=(n,len=2)=>n.toString().padStart(len,'0');
        return `${pad(h,2)}:${pad(m,2)}:${pad(s,2)},${pad(ms,3)}`;
    }

        a.click(); a.remove(); URL.revokeObjectURL(url);
    }

        }
    }

    // Settings persistence and theme
    loadSettings(){
        try{
            const s = localStorage.getItem(this.STORAGE_KEY_SETTINGS);

        // only one modal at a time in this simple system
        this.closeTopModal();
        const backdrop = document.createElement('div');
        backdrop.className = 'modal-backdrop';
        backdrop.tabIndex = -1;
        const modal = document.createElement('div');
        modal.className = 'modal';
        modal.setAttribute('role','dialog');
        modal.setAttribute('aria-modal','true');
        modal.innerHTML = innerHtml;

        // Focus trap
        const focusable = 'a[href], area[href], input:not([disabled]), select:not([disabled]), textarea:not([disabled]), button:not([disabled]), [tabindex]:not([tabindex="-1"])';
        const focusables = modal.querySelectorAll(focusable);
        const firstFocusable = firstFocusSelector ? modal.querySelector(firstFocusSelector) : (focusables[0] || modal);
        const lastFocusable = focusables[focusables.length-1] || firstFocusable;
        setTimeout(()=>{ firstFocusable && firstFocusable.focus(); },30);

            if (e.key==='Tab'){

                else if (!e.shiftKey && document.activeElement === lastFocusable){ e.preventDefault(); firstFocusable.focus(); }
            } else if (e.key === 'Escape'){
                e.stopPropagation(); // let app close it
            }
        }
        modal._keyHandler = handleKey;
        modal.addEventListener('keydown', handleKey);

        // store for later
        document.body._openModal = {backdrop, modal};
    }

    }

    // Audio meter
    async startAudioMeter(){
        try {
            if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) return;
            this.audioStream = await navigator.mediaDevices.getUserMedia({audio:true});
            this.audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            const source = this.audioCtx.createMediaStreamSource(this.audioStream);
            this.analyser = this.audioCtx.createAnalyser();
            this.analyser.fftSize = 256;
            source.connect(this.analyser);
            this.drawVolume();
        } catch(e){
            console.warn('Audio meter unavailable', e);
        }
    }
    stopAudioMeter(){
        if (this.animationId) cancelAnimationFrame(this.animationId);
        if (this.analyser) { this.analyser.disconnect(); this.analyser=null; }
        if (this.audioCtx){ try{ this.audioCtx.close(); }catch(e){} this.audioCtx=null; }
        if (this.audioStream){ this.audioStream.getTracks().forEach(t=>t.stop()); this.audioStream=null; }
        // clear canvas
        const c = this.elements.volumeCanvas;
        if (c && c.getContext){
            const ctx = c.getContext('2d'); ctx.clearRect(0,0,c.width,c.height);
        }
    }
    drawVolume(){
        if (!this.analyser) return;
        const canvas = this.elements.volumeCanvas;
        const ctx = canvas.getContext('2d');
        const bufferLength = this.analyser.frequencyBinCount;
        const dataArray = new Uint8Array(bufferLength);
        const draw = ()=>{
            this.analyser.getByteFrequencyData(dataArray);
            const avg = dataArray.reduce((a,b)=>a+b,0)/dataArray.length;
            // resize canvas for hiDPI
            const dpr = window.devicePixelRatio || 1;
            const width = canvas.clientWidth; const height = canvas.clientHeight;
            canvas.width = width * dpr; canvas.height = height * dpr;
            ctx.clearRect(0,0,canvas.width,canvas.height);
            // draw background bar
            ctx.fillStyle = '#0b1220';
            ctx.fillRect(0,0,canvas.width,canvas.height);
            // draw level
            const level = Math.min(1, avg/160);
            const grad = ctx.createLinearGradient(0,0,canvas.width,0);
            grad.addColorStop(0,'#10b981'); grad.addColorStop(.5,'#f59e0b'); grad.addColorStop(1,'#ef4444');
            ctx.fillStyle = grad;
            ctx.fillRect(0,0,canvas.width * level, canvas.height);
            this.animationId = requestAnimationFrame(draw);
        };
        draw();
    }
}

Limitations / reality-check (I am not going to sugarcoat it):  
• The Web Speech API does not provide per-word timestamps. The SRT generator therefore approximates timestamps from the moment final chunks occur. If you need precise caption timing, you must either record audio and run a timed STT (server-side Whisper, Vosk with forced alignment, or similar), or use a vendor API that returns timestamps.  
• Offline STT (true local) requires adding a local model (Whisper.cpp, Vosk) and/or a server component — not possible in a single static HTML without shipping a heavy WASM model. I can point you to options if you want that route.  
• Browser differences: Chromium-based browsers give the best Web Speech behavior. Safari/Firefox may behave differently.

class SpeechToTextApp {

            this.elements.startBtn.disabled = true;
        } else {
            this.setupRecognition();
        }
    }

    setupRecognition(){
        this.recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
        this.recognition.continuous = true;
        this.recognition.interimResults = true;
        this.updateRecognitionSettings();

        this.recognition.onstart = ()=>{
            this.isListening = true;
            this.updateUIState();
            this.showSuccess("Listening started.");
            this.startAudioMeter();
        };

        this.recognition.onerror = (event)=>{
            console.error('Speech Recognition Error:', event);
            this.showError(`Microphone Error: ${event.error}. Check permissions and try again.`);
            this.stopListening(false);
        };

        this.recognition.onend = ()=>{
            if (this.isListening){
                try { this.recognition.start(); } catch(e){ console.warn('Could not restart recognition:', e); }
            }
        };

        this.recognition.onresult = (event)=>{
            let interim = '';
            let finalChunk = '';
            for (let i=event.resultIndex; i<event.results.length; ++i){
                const transcript = event.results[i][0].transcript.trim();
                if (event.results[i].isFinal){
                    finalChunk += transcript + '. ';
                } else {
                    interim += transcript + ' ';
                }
            }

            if (finalChunk){
                const now = Date.now();
                this.currentSegments.push({text: finalChunk.trim(), ts: now});
                this.currentTranscription += finalChunk;
                this.announceForSR(finalChunk.trim());
            }

            let displayText = this.currentTranscription.trim();
            if (interim) displayText += ' [Interim: ' + interim.trim() + ']';
            this.elements.transcriptionOutput.value = displayText;
            this.elements.transcriptionOutput.scrollTop = this.elements.transcriptionOutput.scrollHeight;
        };
    }

    async startListening(){
        if (!this.recognition) { this.showError("Speech recognition not available."); return; }
        if (this.isListening) { this.showError("Already listening."); return; }

        try {
            await navigator.mediaDevices.getUserMedia({audio:true});
        } catch (e){
            this.showError("Microphone access denied or unavailable.");
            return;
        }

        try {
            this.updateRecognitionSettings();
            this.recognition.start();
            this.isListening = true;
            this.updateUIState();
        } catch (e){
            console.error("Start error:", e);
            this.showError("Failed to start listening.");
        }
    }

            this.currentSegments = [];

        } else if (!saveOnStop){

    convertItemToSrt(item){
        if (item.segments && item.segments.length>0){
            const t0 = item.segments[0].ts;
            const entries = [];
            for (let i=0;i<item.segments.length;i++){
                const seg = item.segments[i];
                const start = Math.max(0, Math.floor((seg.ts - t0)/1000));
                let end;
                if (i+1<item.segments.length) end = Math.max(start+1, Math.floor((item.segments[i+1].ts - t0)/1000));
                else {
                    const approxSec = Math.max(2, Math.ceil(seg.text.length / 20));
                    end = start + approxSec;
                }
                entries.push({start,end,text:seg.text});
            }
            return entries.map((e, i)=>`${i+1}\n${this.formatTimeSrt(e.start)} --> ${this.formatTimeSrt(e.end)}\n${e.text}`).join('\n\n');
        } else {
            const words = item.transcript.split(/\s+/);
            const wordsPerChunk = 10;
            const chunks = [];
            for (let i=0;i<words.length;i+=wordsPerChunk){
                chunks.push(words.slice(i,i+wordsPerChunk).join(' '));
            }
            let sIdx = 0;
            return chunks.map((c, i)=>{
                const start = sIdx;
                const end = sIdx + 4;
                sIdx += 5;
                return `${i+1}\n${this.formatTimeSrt(start)} --> ${this.formatTimeSrt(end)}\n${c}`;
            }).join('\n\n');
        }
    }

    formatTimeSrt(sec){
        const h = Math.floor(sec/3600); const m = Math.floor((sec%3600)/60); const s = sec%60;
        const ms = 0;
        const pad=(n,len=2)=>n.toString().padStart(len,'0');
        return `${pad(h,2)}:${pad(m,2)}:${pad(s,2)},${pad(ms,3)}`;
    }

    openModal(innerHtml, firstFocusSelector){
        this.closeTopModal();
        const backdrop = document.createElement('div');
        backdrop.className = 'modal-backdrop';
        backdrop.tabIndex = -1;
        const modal = document.createElement('div');
        modal.className = 'modal';
        modal.setAttribute('role','dialog');
        modal.setAttribute('aria-modal','true');
        modal.innerHTML = innerHtml;
        backdrop.appendChild(modal);
        document.body.appendChild(backdrop);

        const focusable = 'a[href], area[href], input:not([disabled]), select:not([disabled]), textarea:not([disabled]), button:not([disabled]), [tabindex]:not([tabindex="-1"])';
        const focusables = modal.querySelectorAll(focusable);
        const firstFocusable = firstFocusSelector ? modal.querySelector(firstFocusSelector) : (focusables[0] || modal);
        const lastFocusable = focusables[focusables.length-1] || firstFocusable;
        setTimeout(()=>{ firstFocusable && firstFocusable.focus(); },30);

        function handleKey(e){
            if (e.key==='Tab'){
                if (focusables.length === 0){ e.preventDefault(); return; }
                if (e.shiftKey && document.activeElement === firstFocusable){ e.preventDefault(); lastFocusable.focus(); }
                else if (!e.shiftKey && document.activeElement === lastFocusable){ e.preventDefault(); firstFocusable.focus(); }
            }
        }
        modal._keyHandler = handleKey;
        modal.addEventListener('keydown', handleKey);

        document.body._openModal = {backdrop, modal};
    }

    closeTopModal(){
        const s = document.body._openModal;
        if (s){
            try{
                s.modal.removeEventListener('keydown', s.modal._keyHandler);
            } catch(e){}
            s.backdrop.remove();
            document.body._openModal = null;
        }
    }

    showError(msg){ this.showMessage(msg,'error'); }

    showMessage(msg,type){
        const container = (type==='error')?this.elements.errorContainer:this.elements.successContainer;
        container.innerHTML = '';
        const div = document.createElement('div');
        div.className = (type==='error')? 'error-message':'success-message';
        div.textContent = msg;
        container.appendChild(div);
        if (type==='success') setTimeout(()=>{ if (div.parentNode) div.parentNode.removeChild(div); },4500);
    }

    async startAudioMeter(){
        try {
            if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) return;
            this.audioStream = await navigator.mediaDevices.getUserMedia({audio:true});
            this.audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            const source = this.audioCtx.createMediaStreamSource(this.audioStream);
            this.analyser = this.audioCtx.createAnalyser();
            this.analyser.fftSize = 256;
            source.connect(this.analyser);
            this.drawVolume();
        } catch(e){
            console.warn('Audio meter unavailable', e);
        }
    }
    stopAudioMeter(){
        if (this.animationId) cancelAnimationFrame(this.animationId);
        if (this.analyser) { try{ this.analyser.disconnect(); }catch(e){} this.analyser=null; }
        if (this.audioCtx){ try{ this.audioCtx.close(); }catch(e){} this.audioCtx=null; }
        if (this.audioStream){ this.audioStream.getTracks().forEach(t=>t.stop()); this.audioStream=null; }
        const c = this.elements.volumeCanvas;
        if (c && c.getContext){
            const ctx = c.getContext('2d'); ctx.clearRect(0,0,c.width,c.height);
        }
    }
    drawVolume(){
        if (!this.analyser) return;
        const canvas = this.elements.volumeCanvas;
        const ctx = canvas.getContext('2d');
        const bufferLength = this.analyser.frequencyBinCount;
        const dataArray = new Uint8Array(bufferLength);
        const draw = ()=>{
            this.analyser.getByteFrequencyData(dataArray);
            const avg = dataArray.reduce((a,b)=>a+b,0)/dataArray.length;
            const dpr = window.devicePixelRatio || 1;
            const width = canvas.clientWidth; const height = canvas.clientHeight;
            canvas.width = width * dpr; canvas.height = height * dpr;
            ctx.clearRect(0,0,canvas.width,canvas.height);
            ctx.fillStyle = '#fff';
            ctx.fillRect(0,0,canvas.width,canvas.height);
            const level = Math.min(1, avg/160);
            const grad = ctx.createLinearGradient(0,0,canvas.width,0);
            grad.addColorStop(0,'#10b981'); grad.addColorStop(.5,'#f59e0b'); grad.addColorStop(1,'#ef4444');
            ctx.fillStyle = grad;
            ctx.fillRect(0,0,canvas.width * level, canvas.height);
            this.animationId = requestAnimationFrame(draw);
        };
        draw();
    }
}
