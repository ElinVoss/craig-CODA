import numpy as np, cv2, os, math

def dark_channel(im, size=15):
    # im float [0,1], rgb
    minch = np.min(im, axis=2)
    kernel = cv2.getStructuringElement(cv2.MORPH_RECT,(size,size))
    dark = cv2.erode(minch, kernel)
    return dark

def estimate_atmospheric_light(im, dark, top_percent=0.001):
    # im float [0,1], rgb
    h,w = dark.shape
    num = int(max(1, h*w*top_percent))
    flat = dark.reshape(-1)
    idx = np.argpartition(flat, -num)[-num:]
    # pick the pixel with highest intensity in im among these
    im_flat = im.reshape(-1,3)
    # brightness as sum
    brightest = idx[np.argmax(np.sum(im_flat[idx], axis=1))]
    A = im_flat[brightest]
    return A

def estimate_transmission(im, A, omega=0.95, size=15):
    im_norm = im / (A.reshape(1,1,3) + 1e-6)
    dark = dark_channel(im_norm, size)
    t = 1 - omega*dark
    return t

def recover_radiance(im, A, t, t0=0.1):
    t_clip = np.maximum(t, t0)
    J = (im - A.reshape(1,1,3)) / t_clip[:,:,None] + A.reshape(1,1,3)
    return np.clip(J, 0, 1)

def gray_world_wb(im):
    # im float [0,1], rgb
    mean = np.mean(im.reshape(-1,3), axis=0)
    gray = np.mean(mean)
    scale = gray/(mean+1e-6)
    out = im*scale.reshape(1,1,3)
    return np.clip(out,0,1)

def apply_clahe_lab(im_uint8, clip=2.0, tile=(8,8)):
    lab = cv2.cvtColor(im_uint8, cv2.COLOR_RGB2LAB)
    L,a,b = cv2.split(lab)
    clahe = cv2.createCLAHE(clipLimit=clip, tileGridSize=tile)
    L2 = clahe.apply(L)
    lab2 = cv2.merge([L2,a,b])
    return cv2.cvtColor(lab2, cv2.COLOR_LAB2RGB)

def highlight_compress_lab(im_uint8, knee_start=200, strength=1.0):
    lab = cv2.cvtColor(im_uint8, cv2.COLOR_RGB2LAB)
    L,a,b = cv2.split(lab)
    Lf = L.astype(np.float32)
    ks = knee_start
    above = np.maximum(Lf-ks,0)
    # soft knee compression
    kr = 255-ks
    # strength affects compression aggressiveness
    denom = (kr/(strength+1e-6)) + above
    compressed = ks + above*(kr/denom)
    L2 = np.clip(compressed,0,255).astype(np.uint8)
    lab2 = cv2.merge([L2,a,b])
    return cv2.cvtColor(lab2, cv2.COLOR_LAB2RGB)

def unsharp_mask(im_uint8, amount=1.0, radius=1.2, threshold=0):
    imf = im_uint8.astype(np.float32)
    blurred = cv2.GaussianBlur(imf, (0,0), radius)
    sharp = imf + amount*(imf-blurred)
    sharp = np.clip(sharp,0,255).astype(np.uint8)
    return sharp

def denoise(im_uint8, h=6):
    # fastNlMeansDenoisingColored expects BGR
    bgr = cv2.cvtColor(im_uint8, cv2.COLOR_RGB2BGR)
    out = cv2.fastNlMeansDenoisingColored(bgr, None, h, h, 7, 21)
    return cv2.cvtColor(out, cv2.COLOR_BGR2RGB)

import numpy as np, cv2, os, math

def percentile_wb(img_rgb, low=1, high=99):
    # img_rgb uint8
    out = img_rgb.astype(np.float32)
    for c in range(3):
        lo = np.percentile(out[:,:,c], low)
        hi = np.percentile(out[:,:,c], high)
        if hi - lo < 1:
            continue
        out[:,:,c] = (out[:,:,c] - lo) * (255.0/(hi - lo))
    return np.clip(out,0,255).astype(np.uint8)

def s_curve(x, strength=0.6):
    # x in [0,1]
    # logistic-ish contrast curve with controllable strength
    # strength 0 = identity
    k = 6*strength + 1e-6
    # centered at 0.5
    y = 1/(1+np.exp(-k*(x-0.5)))
    # normalize so 0->0 and 1->1
    y0 = 1/(1+np.exp(k*0.5))
    y1 = 1/(1+np.exp(-k*0.5))
    y = (y - y0)/(y1 - y0)
    return y

def process_clarity(im_rgb):
    # 1) white balance & stretch
    wb = percentile_wb(im_rgb, low=0.5, high=99.2)
    
    # 2) LAB operations
    lab = cv2.cvtColor(wb, cv2.COLOR_RGB2LAB).astype(np.float32)
    L,a,b = cv2.split(lab)
    L_norm = L/255.0
    
    # highlight compression via soft knee on L
    knee = 0.80  # start compress around 80% 
    above = np.maximum(L_norm - knee, 0)
    L_comp = L_norm.copy()
    # compress above knee
    L_comp = np.where(L_norm > knee, knee + above/(1 + 3*above), L_norm)
    
    # shadow lift a bit (gamma)
    gamma = 0.92
    L_lift = np.power(L_comp, gamma)
    
    # apply mild S-curve to add contrast but not blow
    L_sc = (1-0.25)*L_lift + 0.25*s_curve(L_lift, strength=0.55)
    
    # local contrast (clarity) on L
    L_blur = cv2.GaussianBlur(L_sc, (0,0), 12)
    clarity = L_sc + 0.35*(L_sc - L_blur)
    clarity = np.clip(clarity, 0, 1)
    
    L_new = (clarity*255.0)
    lab_new = cv2.merge([L_new, a, b]).astype(np.uint8)
    rgb = cv2.cvtColor(lab_new, cv2.COLOR_LAB2RGB)
    
    # vibrance: boost saturation more on low-sat pixels
    hsv = cv2.cvtColor(rgb, cv2.COLOR_RGB2HSV).astype(np.float32)
    s = hsv[:,:,1]/255.0
    boost = 0.10*(1 - s) + 0.02  # little boost
    hsv[:,:,1] = np.clip((s*(1+boost))*255.0, 0, 255)
    rgb2 = cv2.cvtColor(hsv.astype(np.uint8), cv2.COLOR_HSV2RGB)
    
    # denoise mild
    rgb_dn = denoise(rgb2, h=4)
    # sharpen mild
    rgb_sh = unsharp_mask(rgb_dn, amount=0.9, radius=1.0)
    return rgb_sh

    L_lift=np.power(L_comp,gamma)
    # S curve mild
    L_sc=(1-0.18)*L_lift + 0.18*s_curve(L_lift, strength=0.55)
    # clarity
    sigma=9
    L_blur=cv2.GaussianBlur(L_sc,(0,0),sigma)
    clarity=L_sc + 0.25*(L_sc-L_blur)
    clarity=np.clip(clarity,0,1)
    L_new=(clarity*255.0)
    lab_new=cv2.merge([L_new,a,b]).astype(np.uint8)
    rgb=cv2.cvtColor(lab_new, cv2.COLOR_LAB2RGB)
    # vibrance mild
    hsv=cv2.cvtColor(rgb, cv2.COLOR_RGB2HSV).astype(np.float32)
    s=hsv[:,:,1]/255.0

def luma_unsharp(rgb, amount=1.1, radius=1.0):
    lab = cv2.cvtColor(rgb, cv2.COLOR_RGB2LAB).astype(np.float32)
    L,a,b = cv2.split(lab)
    blur = cv2.GaussianBlur(L, (0,0), radius)
    L2 = L + amount*(L - blur)
    L2 = np.clip(L2, 0, 255)
    lab2 = cv2.merge([L2, a, b]).astype(np.uint8)
    return cv2.cvtColor(lab2, cv2.COLOR_LAB2RGB)
